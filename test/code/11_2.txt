PRECEDENCES = {'sqrt' : 0, 'cos' : 0, 'sin' : 0, 'tan' : 0, 'lg': 0,
            'abs': 0, 'ln' : 0, '(' : 1, ')' : 1, '|' : 2, '!' : 3,
            '~' : 4, '^' : 5, 'mod' : 6, '*' : 7, '/' : 7, '+' : 8,
            '-' : 8}
UNARY = ['sqrt', 'cos', 'sin', 'tan', 'lg', 'ln', '!', '~', 'abs']
class Stack(list):
    """
    List to Stack with push and top
    """
    def top(self):
        """ top """
        return self[- 1]

    def push(self, val):
        """ push """
        self.append(val)



def to_postfix(expression):
    """ convert to postfix """
    express = []
    myvar = Stack()
    get_token = get_tokens(expression)
    for tkn in get_token:
        if tkn[0] == '(':
            myvar.push(tkn)
        if tkn[0] == 'N':
            express.append(tkn)
        if tkn[0] == 'O':
            while len(myvar) > 0:
                if myvar.top()[0] == '(':
                    break
                if PRECEDENCES[myvar.top()[1]] <= PRECEDENCES[tkn[1]]:
                    express.append(myvar.pop())
                else:
                    break
            myvar.push(tkn)
        if tkn[0] == ')':
            while myvar.top()[1] != '(':
                express.append(myvar.pop())
            myvar.pop()
    while len(myvar) > 0:
        express.append(myvar.pop())
    return express

def operate(operation, num1, num2):
    """ binary Calculations """
    if operation == '+':
        return num1 + num2
    if operation == '-':
        return num1 - num2
    if operation == '*':
        return num1 * num2
    if operation == '/':
        return num1 / num2
    if operation == '^':
        return num1 ** num2
    if operation == 'mod':
        return num1 % num2

def get_tokens(expression):
    """ a parser. myvar2 (previous token) helps to decide if - is
    unary or binary """
    opening_bars = expression.count('|') / 2
    tokens = [part for part in re.split(r'([^\\w\\d.])', expression)\\
              if len(part.strip()) > 0]
    myvar2 = True
    for token in tokens:
        if token == '-' and myvar2:
            token = '~'
        if token == '|':
            if opening_bars == 0:
                myvar2 = False
                yield (')', ')')
            else:
                opening_bars -= 1
                myvar2 = True
                yield ('O', 'abs')
                yield ('(', '(')
        elif token == '(':
            myvar2 = True
            yield ('(', token)
        elif token == ')':
            myvar2 = False
            yield (')', token)
        elif token in PRECEDENCES.keys():
            myvar2 = True
            yield ('O', token)
        elif token == 'Pi':
            myvar2 = False
            yield ('N', math.pi)
        elif token == 'e':
            myvar2 = False
            yield ('N', math.e)
        else:
            myvar2 = False
            yield ('N', float(token))

def calc(expression):
    """" Calculator """
    myvar = Stack()
    for tkn in expression:
        if tkn[0] == 'N':
            myvar.push(tkn[1])
        else:
            if tkn[1] in UNARY:
                tnum = myvar.pop()
                myvar.push(unary_op(tkn[1], tnum))
            else:
                tnum2 = myvar.pop()
                tnum1 = myvar.pop()
                myvar.push(operate(tkn[1], tnum1, tnum2))
    result = myvar.pop()
    if round(result, 5) == round(result):
        return int(round(result))
    return round(result, 5)



def unary_op(operation, num):
    """ unay operations """
    if operation in ['sin', 'cos', 'tan']:
        num = math.radians(num)
    opers = {'sqrt': math.sqrt, 'cos': math.cos, 'sin': math.sin,
        'tan': math.tan, 'lg': math.log10 , 'ln': math.log,
        '!': math.factorial, '~': lambda x: x * -1, 'abs': abs}
    result = opers[operation](num)
    return result
