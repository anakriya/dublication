import numpy
def shellsort(a):
  def new_increment(a):
      c = int(len(a) / 2)
      yield c
      while c != 1:
          if c == 2:
              c = 1
          else:
              c = int(numpy.round(c/2.2))
          yield c
  for increment in new_increment(a):
      for c in xrange(increment, len(a)):
          for d in xrange(c, increment-1, -increment):
              if a[d - increment] < a[d]:
                  break
              a[d],a[d - increment] = a[d - increment],a[d]
  return a
